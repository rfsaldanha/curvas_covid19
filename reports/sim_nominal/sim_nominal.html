  summarise(freq = n()) %>%
  ungroup() %>%

    names_from = ano_obito,
    values_from = freq

  rowwise() %>%
  mutate(total = sum(`2020`, `2021`, na.rm = TRUE)) %>%

    col.names = c("Faixa etária", "2020", "2021", "Total")
